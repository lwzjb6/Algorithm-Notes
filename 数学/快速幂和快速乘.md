### 快速幂：
```c++
ll quick_power(int a, int b, int p) {
    ll ans = 1 % p;
    for(; b; b >>= 1){
        if(b & 1) ans = (ll) ans * a % p;
        a = (ll) a * a % p;
    }
    return ans;
}
```


### 快速乘
```c++
// 求解 a * b % p
ll quick_mul(int a, int b, int p) {
    ll ans = 0;
    for(; b; b >>= 1){
        if(b & 1) ans = (ll)(ans + a) % p;
        a = (ll)a * 2 % p;
        // a = (a >> 1) % p;
    }
    return ans;
}
```
